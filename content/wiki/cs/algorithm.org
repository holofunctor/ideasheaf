** Binary Search

Find in [left, right]:

#+begin_src cpp
    template<typename T>
    int binSearch(vector<T> arr, T const &e) {
        int left = 0;
        int right = arr.size() - 1; 

        while (left <= right) {
            int mid = left + (right - left) / 2
            if (arr[mid] == e)
                return mid;
            else if (arr[mid] > e) 
                right = mid - 1;
            else if (arr[mid] < e) 
                left = mid + 1;
        }
        return -1;
    }
#+end_src

*** Variants of Binary Search

The first element equals target:

#+begin_src cpp
    template<typename T>
    int binSearch(vector<T> arr, T const &e) {
        int left = 0;
        int right = arr.size() - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2
            if (arr[mid] >= e) // continue in left subarray when equal
                right = mid - 1;
            else
                left = mid + 1;
        }

        return (left < arr.size() && arr[left] == e) ? left : -1;
    }
#+end_src

The last element equals target:

#+begin_src cpp
    template<typename T>
    int binSearch(vector<T> arr, T const &e) {
        int left = 0;
        int right = arr.size() - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2
            if (arr[mid] > e) 
                right = mid - 1;
            else // continue in right subarray when equal
                left = mid + 1;
        }

        return (right >= 0 && arr[right] == e) ? right : -1;
    }
#+end_src
